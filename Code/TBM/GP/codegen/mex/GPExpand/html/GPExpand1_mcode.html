1function IResult = GPExpand(I,newdim) %#codegen
 2
 3%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 4% Expand an image as per the Gaussian Pyramid.
 5%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 6
 7Wt = [0.0500    0.2500    0.4000    0.2500    0.0500];
 8
 9dim = size(I);
10% newdim = zeros(1,3); 
11% newdim(1) = dim(1); newdim(2) = dim(2); newdim(3) = dim(3);  
12% for i = 1:numel(dim)
13%     if mod(dim(i),2)==0
14%         newdim(i) = dim(i)*2;
15%     else 
16%         newdim(i) = dim(i)*2-1; 
17%     end
18% end
19
20IResult = zeros(newdim,class(I)); % Initialize the array in the beginning ..
21m = [-2:2];n=m;
22
23switch length(dim)
24    case 1
25        %% Pad the boundaries.
26        I = [ I(1) ;  I ;  I(dim(1))];
27        for i = 0 : newdim(1) - 1
28            pixeli = (i - m)/2 + 2;  idxi = find(floor(pixeli)==pixeli);
29            A = I(pixeli(idxi)) .* Wt(m(idxi)+3);
30            IResult(i + 1)= 2 * sum(A(:));
31        end
32    case 2
33        %% Pad the boundaries.
34        I = [ I(1,:) ;  I ;  I(dim(1),:) ];  % Pad the top and bottom rows.
35        I = [ I(:,1)    I    I(:,dim(2)) ];  % Pad the left and right columns.
36        Wt2 = Wt'*Wt;
37
38        for i = 0 : newdim(1) - 1
39            for j = 0 : newdim(2) - 1
40                pixeli = (i - m)/2 + 2;  idxi = find(floor(pixeli)==pixeli);
41                pixelj = (j - m)/2 + 2;  idxj = find(floor(pixelj)==pixelj);
42                A = I(pixeli(idxi),pixelj(idxj)) .* Wt2(m(idxi)+3,m(idxj)+3);
43                IResult(i + 1, j + 1)= 4 * sum(A(:));
44            end
45        end
46    case 3
47        Wt3 = ones(5,5,5);
48        for i = 1:5
49            Wt3(i,:,:) = Wt3(i,:,:) * Wt(i);
50            Wt3(:,i,:) = Wt3(:,i,:) * Wt(i);
51            Wt3(:,:,i) = Wt3(:,:,i) * Wt(i);
52        end
53        
54        %% Pad the boundaries
55        I2 = zeros(dim+2,class(I));
56        I2(2:1+dim(1),2:1+dim(2),2:1+dim(3)) = I;
57        I2(1,:,:)=I2(2,:,:);I2(end,:,:)=I2(end-1,:,:);
58        I2(:,1,:)=I2(:,2,:);I2(:,end,:)=I2(:,end-1,:);
59        I2(:,:,1)=I2(:,:,2);I2(:,:,end)=I2(:,:,end-1);
60        I=I2; %clear I2;
61        
62        for i = 0 : newdim(1) - 1
63            for j = 0 : newdim(2) - 1
64                for k = 0 : newdim(3) - 1
65                    pixeli = (i - m)/2 + 2;  idxi = find(floor(pixeli)==pixeli);
66                    pixelj = (j - m)/2 + 2;  idxj = find(floor(pixelj)==pixelj);
67                    pixelk = (k - m)/2 + 2;  idxk = find(floor(pixelk)==pixelk);
68                    A = I(pixeli(idxi),pixelj(idxj),pixelk(idxk)) .* Wt3(m(idxi)+3,m(idxj)+3,m(idxk)+3);
69                    IResult(i + 1, j + 1,k+1)= 8 * sum(A(:));
70                end
71            end
72        end
73end
